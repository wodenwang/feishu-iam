部署命令：

```bash
git worktree add --detach /tmp/feishu-iam-v1.0.7-build v1.0.7
docker buildx build --platform linux/amd64 -f deploy/api.Dockerfile -t feishu-iam:v1.0.7 --load .
docker save feishu-iam:v1.0.7 -o /tmp/feishu-iam-v1.0.7-linux-amd64.tar
scp /tmp/feishu-iam-v1.0.7-linux-amd64.tar bpmt@120.24.236.92:/home/bpmt/feishu-iam/
ssh bpmt@120.24.236.92 'cd /home/bpmt/feishu-iam && docker load -i feishu-iam-v1.0.7-linux-amd64.tar && FEISHU_IAM_GIT_SYNC=false FEISHU_IAM_PULL_POLICY=never FEISHU_IAM_IMAGE_TAG=v1.0.7 APP_VERSION=1.0.7 GIT_COMMIT=3ab1790 ./upgrade.sh'
```
